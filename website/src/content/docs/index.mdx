<section class="features-modern">
  <div class="section-label">Features</div>
  
  <div class="feature-item">
    <div class="feature-header">
      <span class="feature-marker">01</span>
      <h2>Type Safety</h2>
    </div>
    <p class="feature-description">
      Powerful type system with flow-sensitive type narrowing catches errors at compile-time, not runtime.
    </p>
  </div>

  <div class="feature-item">
    <div class="feature-header">
      <span class="feature-marker">02</span>
      <h2>Optional Types</h2>
    </div>
    <p class="feature-description">
      No more null pointer exceptions! Optional types <code>T?</code> make handling missing values safe and elegant.
    </p>
  </div>

  <div class="feature-item">
    <div class="feature-header">
      <span class="feature-marker">03</span>
      <h2>Error Handling</h2>
    </div>
    <p class="feature-description">
      First-class error handling with error types <code>T ! E</code> that make dealing with failures explicit.
    </p>
  </div>

  <div class="feature-item">
    <div class="feature-header">
      <span class="feature-marker">04</span>
      <h2>Modern Syntax</h2>
    </div>
    <p class="feature-description">
      Clean, expressive syntax that's easy to learn and pleasant to write.
    </p>
  </div>
</section>

<section class="example-modern">
  <div class="section-label">Quick Look</div>
  
```ferret
// Variables with type inference
let name: str = "Ferret";
let version := 1;

// Optional types with type narrowing
let maybeValue: i32? = 42;

if maybeValue != none {
    let doubled: i32 = maybeValue * 2;
}

// Elvis operator for default values
let value: i32 = maybeValue ?: 0;

// Functions
fn greet(name: str) -> str {
    return "Hello, " + name + "!";
}
```
